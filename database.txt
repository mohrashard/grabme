-- WORKERS
CREATE TABLE workers (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  full_name TEXT NOT NULL,
  nic_number TEXT NOT NULL UNIQUE,
  phone TEXT NOT NULL UNIQUE,
  address TEXT,
  emergency_contact TEXT,

  -- Media (Supabase Storage)
  nic_front_url TEXT,
  nic_back_url TEXT,
  profile_photo_url TEXT,
  selfie_url TEXT,
  past_work_photos TEXT[],
  certificate_url TEXT,

  -- Qualifications
  trade_category TEXT NOT NULL,
  sub_skills TEXT[],
  years_experience INTEGER,
  short_bio TEXT,

  -- Location
  districts_covered TEXT[],

  -- Social Proof
  reference_name TEXT,
  reference_phone TEXT,
  previous_employer TEXT,

  -- Accountability
  agreed_to_code_of_conduct BOOLEAN DEFAULT FALSE,

  -- Pipeline Status
  account_status TEXT DEFAULT 'pending',
  -- Values: 'pending' | 'whatsapp_pinged' |
  --         'under_review' | 'active' |
  --         'rejected' | 'suspended'

  -- Trust Badges (manually flipped by founder)
  is_identity_verified BOOLEAN DEFAULT FALSE,
  is_reference_checked BOOLEAN DEFAULT FALSE,
  is_certificate_verified BOOLEAN DEFAULT FALSE,

  -- Timestamps
  whatsapp_pinged_at TIMESTAMP WITH TIME ZONE,
  activated_at TIMESTAMP WITH TIME ZONE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- VERIFICATION LOG (founder's internal notes)
CREATE TABLE verification_log (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  worker_id UUID REFERENCES workers(id),
  action TEXT,
  -- 'nic_checked' | 'reference_called' |
  -- 'approved' | 'rejected' | 'suspended'
  outcome TEXT,
  -- 'pass' | 'fail' | 'no_answer' | 'suspicious'
  admin_notes TEXT,
  nic_checked BOOLEAN DEFAULT FALSE,
  reference_called BOOLEAN DEFAULT FALSE,
  checked_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
